    <script>
        // Global State variables
        let L = 25.0; // Max bunch weight (kg)
        let k = 0.35; // Growth rate constant
        let t0 = 11.0; // Inflection week
        let selectedWeek = 11.0;
        let viewMode = 'weight'; // 'weight' or 'combined'
        let chart = null;

        // Model evaluation functions
        function calculateWeight(t, currentL = L, currentK = k, currentT0 = t0) {
            return currentL / (1 + Math.exp(-currentK * (t - currentT0)));
        }

        function calculateDerivative(t, currentL = L, currentK = k, currentT0 = t0) {
            const expTerm = Math.exp(-currentK * (t - currentT0));
            return (currentL * currentK * expTerm) / Math.pow(1 + expTerm, 2);
        }

        // Biological explanation generator depending on weeks
        function getBiologicalDetails(t, dW) {
            if (t < 5) {
                return {
                    phase: "LAG / CELL DIVISION",
                    color: "bg-blue-500 text-white",
                    desc: `Weeks 0 to 5 represent early growth stage. The derivative is low ($W'({${t.toFixed(1)}}) = {${dW.toFixed(2)}} \\text{ kg/week}$). Here, the tree prioritizes cell division rather than nutrient storage. Focus on watering and preventing insect damage during this delicate post-fertilization cycle.`
                };
            } else if (t >= 5 && t < 10) {
                return {
                    phase: "LINEAR BULKING",
                    color: "bg-teal-500 text-slate-950",
                    desc: `At Week ${t.toFixed(1)}, cellular expansion starts accelerating. The derivative is rising steadily ($W'({${t.toFixed(1)}}) = {${dW.toFixed(2)}} \\text{ kg/week}$). Water intake and nitrogen application is critical now to ensure optimal kernel development inside each fruitlet.`
                };
            } else if (t >= 10 && t <= 13) {
                return {
                    phase: "INFLECTION / PEAK GROWTH SPEED",
                    color: "bg-amber-500 text-slate-950",
                    desc: `Maximum physiological development is active! The derivative has peaked at $W'({${t.toFixed(1)}}) = {${dW.toFixed(2)}} \\text{ kg/week}$. Under standard MPOB conditions, the plant is consuming large volumes of potassium (K). This inflection point dictates the ultimate bunch structural capacity.`
                };
            } else if (t > 13 && t < 18) {
                return {
                    phase: "MESOCARP OIL SYNTHESIS",
                    color: "bg-orange-500 text-white",
                    desc: `Week ${t.toFixed(1)} represents active oil synthesis. The total bunch weight continues climbing, but the derivative begins a downward trend ($W'({${t.toFixed(1)}}) = {${dW.toFixed(2)}} \\text{ kg/week}$). Plant cells shift from building volume to converting sugars into valuable lipids inside the mesocarp tissue.`
                };
            } else {
                return {
                    phase: "MATURITY / HARVEST WINDOW",
                    color: "bg-green-600 text-white",
                    desc: `At Week ${t.toFixed(1)}, the crop is fully mature. The derivative is near zero ($W'({${t.toFixed(1)}}) = {${dW.toFixed(2)}} \\text{ kg/week}$), indicating that biomass gains have ceased. Harvesting now guarantees a maximum Oil Extraction Rate (OER) before fruitlet detachment and loss starts occurring.`
                };
            }
        }

        // Construct Dataset Arrays for Chart.js
        function generateChartData() {
            const weeksArray = [];
            const weightArray = [];
            const derivativeArray = [];

            for (let i = 0; i <= 22; i += 0.5) {
                weeksArray.push(i);
                weightArray.push(calculateWeight(i));
                derivativeArray.push(calculateDerivative(i));
            }

            // Tangent Line Calculation: Y - Y1 = m(X - X1)  =>  Y = m(X - X1) + Y1
            const selectedY = calculateWeight(selectedWeek);
            const selectedM = calculateDerivative(selectedWeek);
            
            const tangentArray = weeksArray.map(w => {
                // Keep the tangent line local near selected week to prevent confusing visuals
                if (Math.abs(w - selectedWeek) > 3.5) return null;
                return selectedM * (w - selectedWeek) + selectedY;
            });

            return {
                labels: weeksArray,
                weight: weightArray,
                derivative: derivativeArray,
                tangent: tangentArray,
                selectedPoint: [{ x: selectedWeek, y: selectedY }]
            };
        }

                                    let label = context.dataset.label || '';
                                    if (label) {
                                        label += ': ';
                                    }
                                    if (context.parsed.y !== null) {
                                        label += context.parsed.y.toFixed(2);
                                    }
                                    return label;
                                }
                            }
                        }
                    },
                    onClick: (e) => {
                        const points = chart.getElementsAtEventForMode(e, 'nearest', { intersect: false }, true);
                        if (points.length > 0) {
                            const pointIndex = points[0].index;
                            const weekVal = data.labels[pointIndex];
                            updateDashboard(weekVal);
                            updateChart();
                        }
                    }
                }
            });
        }

        // Rapid partial redraw of elements to boost interactive speed
        function updateChart() {
            if (!chart) return;
            const newData = generateChartData();
            
            // Weight dataset
            chart.data.datasets[0].data = newData.weight;
            // Tangent dataset
            chart.data.datasets[1].data = newData.tangent;
            // Highlight spot dataset
            chart.data.datasets[2].data = [{x: selectedWeek, y: calculateWeight(selectedWeek)}];

            if (viewMode === 'combined' && chart.data.datasets[3]) {
                chart.data.datasets[3].data = newData.derivative;
            }

            chart.update('none'); // silent update prevents annoying animations while dragging
        }

        // Setup User Events & Interaction Listeners
        function setupEventListeners() {
            sliderL.addEventListener('input', (e) => {
                L = parseFloat(e.target.value);
                valL.textContent = `${L.toFixed(1)} kg`;
                updateDashboard(selectedWeek);
                updateChart();
            });

            sliderK.addEventListener('input', (e) => {
                k = parseFloat(e.target.value);
                valK.textContent = k.toFixed(2);
                updateDashboard(selectedWeek);
                updateChart();
            });

            sliderT0.addEventListener('input', (e) => {
                t0 = parseFloat(e.target.value);
                valT0.textContent = `${t0.toFixed(1)} weeks`;
                updateDashboard(selectedWeek);
                updateChart();
            });

            // Toggle view modes
            viewWeightBtn.addEventListener('click', () => {
                viewMode = 'weight';
                viewWeightBtn.className = "px-3 py-1.5 rounded-md font-semibold bg-white text-slate-800 shadow-sm transition";
                viewCombinedBtn.className = "px-3 py-1.5 rounded-md font-semibold text-slate-600 hover:text-slate-800 transition";
                initChart();
            });

            viewCombinedBtn.addEventListener('click', () => {
                viewMode = 'combined';
                viewCombinedBtn.className = "px-3 py-1.5 rounded-md font-semibold bg-white text-slate-800 shadow-sm transition";
                viewWeightBtn.className = "px-3 py-1.5 rounded-md font-semibold text-slate-600 hover:text-slate-800 transition";
                initChart();
            });

            // Reset application params to default
            resetBtn.addEventListener('click', () => {
                L = 25.0;
                k = 0.35;
                t0 = 11.0;
                selectedWeek = 11.0;

                sliderL.value = L;
                sliderK.value = k;
                sliderT0.value = t0;

                valL.textContent = "25.0 kg";
                valK.textContent = "0.35";
                valT0.textContent = "11.0 weeks";

                updateDashboard(selectedWeek);
                updateChart();
            });
        }

        // Application bootstrap
        window.onload = function() {
            updateDashboard(selectedWeek);
            initChart();
            setupEventListeners();
        };
    </script>
